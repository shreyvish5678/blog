function createBox(origamiPaper):
	foldHoriz(origamiPaper)
	foldVert(origamiPaper)
    repeat for each corner:
        foldCorner(origamiPaper)
    repeat for each end:
        repeat for each side:
            foldIntoCenter(origamiPaper)
            open(origamiPaper)
            openFlap(origamiPaper)
        repeat for each side:
            foldDownCorner(origamiPaper)
            open(origamiPaper)
    repeat for each end:
        openFlap(origamiPaper)
        foldVert(origamiPaper)
    if wantLid is true:
        return origamiPaper + createBox(origamiPaper)
    else:
        return origamiPaper
